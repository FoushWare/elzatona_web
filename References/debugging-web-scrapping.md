    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // Only show on blog pages
            const isBlogPage = window.location.pathname.includes('/blog/') && !window.location.pathname.includes('/blog/create') && !window.location.pathname.includes('/blog/edit');

            if (!isBlogPage) return;

            // Check if user already interacted with popup
            const popupDismissed = localStorage.getItem('blog_popup_dismissed');
            const userSubscribed = localStorage.getItem('blog_subscribed');

            if (popupDismissed || userSubscribed) return;

            const popup = document.getElementById('blog-popup-overlay');
            const closeBtn = document.getElementById('close-blog-popup');
            const noThanksBtn = document.getElementById('no-thanks-popup');
            const form = document.getElementById('popup-newsletter-form');
            const message = document.getElementById('popup-form-message');

            // Show popup after 30 seconds
            setTimeout(function() {
                if (!localStorage.getItem('blog_popup_dismissed') && !localStorage.getItem('blog_subscribed')) {
                    popup.classList.remove('hidden');
                }
            }, 30000); // 30 seconds

            // Close popup handlers
            function closePopup() {
                popup.classList.add('hidden');
                localStorage.setItem('blog_popup_dismissed', 'true');
            }

            closeBtn.addEventListener('click', closePopup);
            noThanksBtn.addEventListener('click', closePopup);

            // Close on overlay click
            popup.addEventListener('click', function(e) {
                if (e.target === popup) {
                    closePopup();
                }
            });

            // Form submission
            form.addEventListener('submit', function(e) {
                e.preventDefault();

                const formData = new FormData(form);
                const email = formData.get('email');
                const source = formData.get('source');

                message.classList.remove('hidden', 'text-green-600', 'text-red-600');
                message.textContent = 'Subscribing...';
                message.classList.add('text-gray-600');

                fetch('/api/subscribe', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]').getAttribute('content')
                    },
                    body: JSON.stringify({
                        email: email,
                        source: source
                    })
                })
                .then(response => response.json())
                .then(data => {
                    message.textContent = data.message;

                    if (data.success) {
                        message.classList.remove('text-gray-600', 'text-red-600');
                        message.classList.add('text-green-600');
                        form.reset();
                        localStorage.setItem('blog_subscribed', 'true');

                        // Close popup after success
                        setTimeout(function() {
                            popup.classList.add('hidden');
                        }, 2000);
                    } else {
                        message.classList.remove('text-gray-600', 'text-green-600');
                        message.classList.add('text-red-600');
                    }
                })
                .catch(error => {
                    message.textContent = 'An error occurred. Please try again.';
                    message.classList.remove('text-gray-600', 'text-green-600');
                    message.classList.add('text-red-600');
                    console.error('Error:', error);
                });
            });
        });
    </script>
